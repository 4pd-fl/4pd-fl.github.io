[docs]def response_decorator(func):
    @functools.wraps(func)
    def new_f(*args, **kwgs):
        resp = func(*args, **kwgs)
        resp.raise_for_status()
        return resp.json()

    return new_f




[docs]def print_response(res, only_return=True):
    res_content = 'HTTP/1.1 {status_code}\n{headers}\n{body}'.format(
        status_code=res.status_code,
        headers='\n'.join('{}: {}'.format(k, v) for k, v in res.headers.items()),
        body=res.text
    )
    req = res.request
    req_conent = 'HTTP/1.1 {method} {url}\n{headers}\n\n'.format(
        method=req.method,
        url=req.url,
        headers='\n'.join('{}: {}'.format(k, v) for k, v in req.headers.items())
    )
    if 'multipart/form-data' not in req.headers.get('content-type', ''):
        req_conent += (str(req.body) or ' ')
    content = '\n'.join([req_conent, res_content])
    if only_return:
        return content
    else:
        log.info(content)
