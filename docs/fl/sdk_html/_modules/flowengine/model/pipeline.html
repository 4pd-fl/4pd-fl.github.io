[docs]    def edges_exist(self, src_node_id, src_slot_id, dest_node_id, dest_slot_id):
        key = '{}_{}_{}_{}'.format(src_node_id, src_slot_id, dest_node_id, dest_slot_id)
        if self.edges.get(key) != None:
            return True
        else:
            return False



[docs]    def add_edge_raw(self, src_node_id, src_slot_id, dest_node_id, dest_slot_id):
        if not self.edges_exist(src_node_id, src_slot_id, dest_node_id, dest_slot_id):
            edge = Edge(src_node_id, src_slot_id, dest_node_id, dest_slot_id)
            self.edges['{}_{}__{}_{}'.format(src_node_id, src_slot_id, dest_node_id, dest_slot_id)] = edge
            output_slot = OutputSlot(src_slot_id)
            self.nodes[src_node_id].output_slots.append(output_slot)
            input_slot = InputSlot(dest_slot_id,"input_{}".format(dest_slot_id))
            self.nodes[dest_node_id].input_slots.append(input_slot)
            if not self.nodes[dest_node_id].node.exists_param(dest_slot_id):
                self.nodes[dest_node_id].node.params.append(Param(id=dest_slot_id,type = 'TEXT',key="input_{}".format(dest_slot_id),value="{}"))
            self.nodes[dest_node_id].is_start_node = False



[docs]    def add_edge(self, src_node_id, dest_node_id):
        src_node = self.nodes[src_node_id]
        output_slot_id = 'osid_{}_0'.format(src_node_id)
        dest_node = self.nodes[dest_node_id]
        input_slot_id = None
        if len(dest_node.input_slots) > 0:
            input_slot_id = 'isid_{}_{}'.format(dest_node_id, len(dest_node.input_slots))
        else:
            input_slot_id = 'isid_{}_0'.format(dest_node_id)
        self.add_edge_raw(src_node_id, output_slot_id, dest_node_id, input_slot_id)



[docs]    def get_flow_engine_pipeline(self):
        pipeline = {}
        f_nodes = []
        for node in self.nodes.values():
            f_node = self.translate_to_fnode(node)
            f_nodes.append(f_node)
        f_edges = []
        for edge in self.edges.values():
            f_edge = self.translate_fedges(edge)
            f_edges.append(f_edge)

        self.append_start_node_to_fpipeline(f_edges, f_nodes)
        execution_config = {}
        execution_config['runMode'] = self.execution_config.run_mode
        execution_config['cronExpression'] = self.execution_config.cron_expression
        execution_config['interval'] = self.execution_config.interval
        execution_config['intervalUnit'] = self.execution_config.interval_unit
        execution_config['trigger'] = self.execution_config.trigger
        execution_config['retryCount'] = self.execution_config.retry_count
        execution_config['delayTime'] = self.execution_config.delay_time
        execution_config['delayTimeUnit'] = self.execution_config.delay_time_unit

        pipeline['id'] = self.pipeline_id
        pipeline['name'] = self.pipeline_name
        pipeline['nodes'] = f_nodes
        pipeline['edges'] = f_edges
        pipeline['executionConfig'] = execution_config
        return pipeline



[docs]    def append_start_node_to_fpipeline(self, edges, f_nodes):
        start_node = self.get_start_node()
        f_nodes.append(start_node)
        for node in self.nodes.values():
            if node.is_start_node:
                f_edge = {}
                f_edge['srcNodeId'] = start_node['nodeId']
                f_edge['destNodeId'] = node.node_id
                f_edge['srcSlotId'] = start_node['outputSlots'][0]['id']
                f_edge['destSlotId'] = 'isid_{}_0'.format(node.node_id)
                f_edge['id'] = '{}'.format(uuid.uuid1())
                edges.append(f_edge)
                input_slot = InputSlot(id='isid_{}_0'.format(node.node_id))
                node.input_slots.append(input_slot)



[docs]    def translate_fedges(self, edge):
        f_edge = {}

        f_edge['destNodeId'] = edge.dest_node_id
        f_edge['srcSlotId'] = edge.src_slot_id
        f_edge['destSlotId'] = edge.dest_slot_id

[docs]    def translate_to_fnode(self, node):
        f_node = {}
        f_node['name'] = node.node.job_name
        if node.node.index_type == "PythonJobRunner":
            # f_node['indexType'] = "PythonJobRunner"
            # index_source = {}
            # index_source['runnerPath'] = settings['python_runner_path']
            # index_source['jobTemplate'] = node.node.job_template
            args = node.node._args
            jobArgs = {}
            for arg in args:
                jobArgs[arg] = getattr(node.node, arg, None)
            jobArgsStr =  json.dumps(jobArgs,default=lambda o: o.__dict__,sort_keys=True)
            #index_source['jobArgs'] = jobArgs
           # f_node['indexSource'] = json.dumps(index_source,default=lambda o: o.__dict__,sort_keys=True,indent=6);

            f_node['indexType'] = 'SHELL'
            f_node['indexSource'] = "ml_cli job runner {jobTemplate} {jobArgs}".format(jobTemplate=node.node.job_template,jobArgs=json.dumps(jobArgsStr))
        else:
            f_node['indexType'] = node.node.index_type
            f_node['indexSource'] = node.node.index_source
        f_node['params'] = {'values':node.node.params}
        outputConfig = {}
        if node.node.index_type == "DAG":
            outputConfig['outPutNodeId'] = node.node.output_config.output_node_id
            outputConfig['outputType'] = node.node.output_config.output_type
        else:
            outputConfig['outPutNodeId'] = None
            outputConfig['outputType'] = "Table"
            outputConfig['outPutFileKey'] = "outputFile"
        f_node['outputConfig'] = outputConfig
        f_node['type'] = node.node_type
        f_node['status'] = 'INITIALIZED'
        f_node['nodeId'] = node.node_id
        f_node['title'] = node.node.job_name
        f_node['inputSlots'] =  node.input_slots
        f_node['outputSlots'] = node.output_slots
        f_node['isStartNode'] = node.is_start_node
        if node.node_type == 'SENSOR':
            f_node['sensorJobConfig'] = node.node.sensor_config
        f_node['executionConfig'] = node.node.execution_config
        f_node['openNodeMsgTopicConfig'] = node.node.msgPublishConfigs
        return f_node
