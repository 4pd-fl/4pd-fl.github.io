[docs]class ExecutionConfig:
    def __init__(self,
                 run_mode= RunMode.SINGLE.value,  # SINGLE,PERIODICAL,CRONTAB,MESSAGE
                 cron_expression: Optional[str] = '',  # CRONTAB表达式 runMode = CRONTAB ,
                 interval: Optional[str] = 0,  # 周期任务模式的间隔时间 runMode = PERIODICAL，
                 interval_unit: Optional[str] = TimeUnit.MINUTE.value,  # 周期任务模式的间隔单位，HOUR,DAY,MINUTE,SECOND  runMode = PERIODICAL，
                 trigger: Optional[str] = '',  # msg模式下触发的topic runMode = MESSAGE，
                 retry_count: Optional[int] = 1,  # job失败重试次数
                 delay_time: Optional[int] = 0,  # sensor模式下延时触发 1.5.2 暂不支持保留字段
                 delay_time_unit: Optional[str] = TimeUnit.MINUTE.value  # HOUR,DAY,MINUTE,SECOND 1.5.2 暂不支持保留字段
                 ):
        self.run_mode = run_mode
        self.cron_expression = cron_expression
        self.interval = interval
        self.interval_unit = interval_unit
        self.trigger = trigger
        self.retry_count = retry_count
        self.delay_time = delay_time
        self.delay_time_unit = delay_time_unit
