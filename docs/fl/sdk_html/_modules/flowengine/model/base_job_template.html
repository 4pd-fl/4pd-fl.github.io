    def __init__(self,
                 job_id: str,
                 job_name: Optional[str] = None,
                 params: Optional[List[Param]] = None,
                 output_config: Optional[OutputConfig] = OutputConfig(),
                 execution_config:Optional[ExecutionConfig] = ExecutionConfig(),
                 msgPublishConfigs:Optional[List[MsgPublishConfig]] = list(),
                 pipeline: Optional[Pipeline] = None,
                 ):
        self.log = logger.getLogger(__name__)
        self.job_id = job_id
        if job_name != None:
            self.job_name = job_name
        else:
            self.job_name = self.__class__.__name__
        self.job_template = "{}.{}".format(self.__class__.__module__, self.__class__.__name__)
        if params == None:
            params =list()
        self.params = params
        self.type="TASK"
        self.output_config = output_config
        self.execution_config = execution_config
        self.msgPublishConfigs = msgPublishConfigs
        self._args = ['job_id', 'job_name', 'params', 'execution_config','output_config']
        self.add_to_pipeline(pipeline)

[docs]    def add_to_pipeline(self, pipeline):

        if pipeline is not None and self.job_id not in pipeline.nodes:
            pipeline.add_node(self, self.type)

        self.__pipeline = pipeline


    def __rshift__(self, other):
        """
        Implements Self >> Other == self.set_downstream(other)
        """
        self.set_downstream(other)
        return other

    def __lshift__(self, other):
        """
        Implements Self << Other == self.set_upstream(other)
        """
        self.set_upstream(other)
        return other

    def __rrshift__(self, other):
        """
        Called for [DAG] >> [Operator] because DAGs don't have
        __rshift__ operators.
        """
        self.__lshift__(other)
        return self

    def __rlshift__(self, other):
        """
        Called for [DAG] << [Operator] because DAGs don't have
        __lshift__ operators.
        """
        self.__rshift__(other)
        return self
