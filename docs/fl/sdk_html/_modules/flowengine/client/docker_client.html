[docs]    def save_image(self, image_url, target_file_name):
        print(colored('Downloading ' + target_file_name, 'green'))
        self.client.pull(image_url, stream=False, decode=True)
        self.client.images()
        image = self.client.get_image(image_url)
        f = open(target_file_name, 'wb')
        for chunk in image:
            f.write(chunk)
        f.close()
