    def _get_by_prn(self, url, prn):
        url = "{}?prn={}".format(url, prn)
        return requests.get(url, headers=self.headers)

    def _create_using_json(self, url, obj, upsert=None):
        if upsert is None:
            pass
        else:
            url = '{}?upsert={}'.format(url, upsert)
        body = obj if type(obj) == dict else obj.toDict()
        return requests.post(url, None, body, headers=self.headers)
