from flowengine.client.base_client import BaseClient



[docs]class MarketClient(BaseClient):

    def __init__(self, url, headers=None):
        super().__init__()
        self.headers = headers
        self.headers.pop('content-type')
        self.url = url
        self.market_url = url + '/template-market/v1'
        self.asol_url = self.market_url + '/asol'
        self.sc_url = self.market_url + '/component'


[docs]    def prepare_sc(self, sc_config):
        url = '{}/prepare'.format(self.sc_url)
        return requests.post(url, json=sc_config, headers=self.headers)



[docs]    def upload_solution_ui(self, key, version, group, file_path):
        url = '{url}/uploadSolutionUi'.format(url=self.sc_url)
        attachment = {'file': open(file_path, 'rb')}
        sc_config = {
            'key': key,
            'version': version,
            'group': group
        }
        return requests.post(url, data=sc_config, files=attachment, headers=self.headers, verify=False)

        return requests.post(url, json=body, headers=self.headers, verify=False)
