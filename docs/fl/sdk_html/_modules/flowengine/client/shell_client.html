    def _sudo_cmd(cmd, password):
        return 'echo {password} | sudo -S {cmd}'.format(password=password, cmd=cmd)

    def run_yarn(cmd, sudo=False, password=''):
        """执行 yarn 命令

        执行 yarn 命令

        Args:
            :cmd（str):需要执行的命令行命令
            :sudo (bool): 默认否，是否使用 sudo 执行命令
            :passwor (str): 默认空，sudo 用户的明文密码

        Returns:
            :p (Popen):执行命令对象
            :app_id (str): yarn app 的 id
            :tracking_ui (str): yarn app 的 tracking ui 链接
            :retcode (int): 命令执行后的返回值，例如：0 表示执行正常，1 表示执行出错。
        """
        if sudo:
            assert password != None and len(password) > 0, 'password is needed when sudo is true.'
            cmd = ShellClient._sudo_cmd(cmd, password)
        retcode = None
        p = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.STDOUT, shell=True)
        app_id = None
        tracking_ui = None
        for __ in range(1000):
            retcode = p.poll()
            line = p.stdout.readline()
            line = line.decode("utf-8")
            log.info(line)
            if 'appTrackingUrl' in line:
                tracking_ui = line.split()[-1]
            elif 'appId' in line:
                app_id = line.split()[-1]
            if tracking_ui is not None and app_id is not None:
                break
            if retcode is not None:
                break
        return p, app_id, tracking_ui, retcode
