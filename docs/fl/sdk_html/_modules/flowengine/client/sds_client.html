[docs]    def import_hdfs_file(self, table_name, hdfs_path, project_id):
        # get source data info
        data_url = '{projectId}/sourceDatas/byCondition'.format(projectId=project_id)
        data_body = {
            'projectId': project_id,
            'name': table_name
        }
        data_result = self.request(data_url, 'post', body=data_body)
        source_data = data_result.json()['data'][0]

        source_data_id = source_data['id']
        source_schema = source_data['groupSchema']

        url = '{projectId}/sourceDatas/{id}/hdfs/import'.format(projectId=project_id, id=source_data_id)
        body = {
            'dataSource': {
                'url': hdfs_path,
                "protocol": "HDFS",
                "fileType": "parquet",
                "lineSpliter": "0x0A",
                "columnSpliter": ",",
                "encoding": "utf-8",
                "isFirstLineSchema": True,
                "nullValues": []
            },
            'dataTarget': {
                "typeError": "SKIP",
                "columnError": "SKIP",
                "schema": source_schema,
                "format": "parquet",
                "isOverwrite": False,
                "prn": "hdfs"
            },
            'config': {
                "advanced_config": "{\"-m\":\"4\",\"mapreduce.job.reduces\":\"3\",\"mapreduce.map.memory.mb\":\"4096\",\"mapreduce.map.java.opts\":\"-Xmx3072M\",\"mapreduce.reduce.memory.mb\":\"4096\",\"mapreduce.reduce.java.opts\":\"-Xmx3072M\"}"
            }
        }
        return self.request(url, 'post', body=body)
