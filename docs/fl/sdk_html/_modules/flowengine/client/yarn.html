[docs]    def kill_app(self, app_id):
        assert (app_id is not None)
        url = '%s/ws/v1/cluster/apps/%s/state?user.name=%s' % (self.url,
                                                               app_id,
                                                               self.user_name)
        body = dict(state="KILLED")
        kwargs = dict(body=body)
        return requests.put(url, None, kwargs, headers=self.headers)
