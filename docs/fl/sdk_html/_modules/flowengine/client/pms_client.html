[docs]    def create_alert(self, body):
        """创建告警

        :param body

        [
            {
                "labels":{
                    "alertname":"watchdog_1_4",
                    "customer":"xx",
                    "instance":"xx.xx.xx:80",
                    "job":"watchdog_exporter"
                },
                "annotations":{
                    "description":"Watchdog: xxx-managementwin",
                    "summary":"Watchdog: xxn, UCM: xxxx"
                },
                "startsAt":"0001-01-01T00:00:00Z",
                "endsAt":"2018-01-31T21:49:06.661888851+01:00",
                "generatorURL":"http://prometheus11.xxx.xxx:9090"
            },
            ...
        ]

        labels

        事件标签词典，其中 alertname 是必须的，其它 key 是自定义和可选的； key 和 value 必须都是字符串格式；

        annotations

        事件详情词典，可选，key 是自定义的；key 和 value 必须都是字符串格式；

        startsAt

        告警开始时间；可选，如果未填则为当前时间；

        endsAt

        告警结束时间；可选，如果 endsAt < 当前时间，则表示告警结束；

        generatorURL

        告警关联的 URL，可选，点击可以查看告警详情；


        :return:

        {
          "count": 0,
          "status": "string"
        }
        """

        url = "{alerts_url}".format(alerts_url=self.ALERTS_URL)
        return requests.post(url, json=body, headers=self.headers)

[docs]    def create_alert_policy(self, body):
        """创建告警规则

        :param body

        name	告警发送策略名称，不能重复
        routes	告警匹配规则
        routes.receiver
        接收人名称，必须在 receivers 中定义

        routes.group_by	可选，聚合告警的标签名称列表
        routes.group_wait	可选，收到告警后等待指定时间
        routes.group_interval	可选，按照指定的时间间隔对告警进行分组聚合
        routes.repeat_interval	可选，发送告警后，再等待指定时间间隔发送告警
        match.re	匹配告警中对应 label 的正则表达式列表
        receivers
        告警接收人列表，目前只支持 email 和 webhook 两种方式；

        接收人名称不能重复！

        receivers.name	接收人名称
        receivers.email_configs	接收人对应的 email 配置参数
        receivers.webhook_configs	接收人对应的 webhook 配置参数
        email_configs 和 webhook_configs 其它参数，参考：https://prometheus.io/docs/alerting/configuration/



        参数示例：

        {
            "name": "testAlertPolicy",
            "routes": [
                {
                    "receiver": "cloudhosting",
                    "match_re": {
                        "alertname": "^watchdog$",
                        "team": "^cloudhosting$"
                    },
                    "group_by": [],
                    "group_wait": "1s",
                    "group_interval": "1s",
                    "repeat_interval": "1s"
                }
            ],
            "receivers": [
                {
                    "name": "cloudhosting",
                    "email_configs": [
                        { "to": "xx@xx.com"}
                    ]
                },
                {
                    "name": "hooker",
                    "webhook_configs": [
                        { "url": "http://path/to/web/hook"}
                    ]
                }
            ]
        }

        :return:

        """

        url = "{alert_policy_url}".format(alert_policy_url=self.ALERTPOLICY_URL)
        return requests.post(url, json=body, headers=self.headers)
