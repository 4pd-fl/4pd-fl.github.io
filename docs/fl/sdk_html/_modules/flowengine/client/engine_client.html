    def __init__(self, url, headers=None):

        self.headers = headers
        self.server_url = url
        self.manager_url = self.server_url + '/automl-manager/v1'
        self.keystone_url = self.server_url + '/keystone/v1'
        self.market_url = self.server_url + '/template-market/v1'
