[docs]    def exec_cmd(self, cmd, pod, name_space):
        try:
            self.api.read_namespaced_pod(name=pod,
                                         namespace=name_space)
        except ApiException as e:
            if e.status != 404:
                print("Unknown error: %s" % e)
                exit(1)

        exec_command = [
            '/bin/bash',
            '-c',
            cmd]
        resp = self.api.connect_get_namespaced_pod_exec(pod, name_space,
                                                        command=exec_command,
                                                        stderr=True, stdin=False,
                                                        stdout=True, tty=False)
        return resp



[docs]    def exec(self, pod, name_space):
        try:
            self.api.read_namespaced_pod(name=pod,
                                         namespace=name_space)
        except ApiException as e:
            if e.status != 404:
                print("Unknown error: %s" % e)
                exit(1)
        exec_command = ['/bin/bash']
        resp = self.api.connect_get_namespaced_pod_exec(pod, name_space,
                                                        command=exec_command,
                                                        stderr=True, stdin=True,
                                                        stdout=True, tty=True,
                                                        _preload_content=False)
        promote = None
        while resp.is_open():
            resp.update(timeout=1)
            if resp.peek_stdout():
                promote = resp.read_stdout()
                promote = promote.strip().strip('\n')
                print(promote, end='', flush=True)
                break
        while resp.is_open():
            x = input()
            x = x.strip()
            if x == 'exit':
                break
            resp.write_stdin(x + '\n')
            while True:
                if resp.peek_stdout():
                    content = resp.read_stdout()
                    # content = filter_nonprintable(content)
                    content = content.strip()
                    content = content.strip('\n')
                    content = content.replace(x, '')
                    # if len(content) > 0:
                    #     print('content:%s..' % content.encode('utf-8'))
                    if promote in content or content.endswith('#'):
                        print(content, end='', flush=True)
                        break
                    elif len(content) > 0:
                        print(content, flush=True)
                if resp.peek_stderr():
                    print(resp.read_stderr(), flush=True)
                resp.update(timeout=1)

        resp.close()



[docs]    def logs(self, pod, name_space, follow=False):
        config.load_kube_config()
        configuration.assert_hostname = False
        api = core_v1_api.CoreV1Api()

        resp = api.read_namespaced_pod_log(pod, namespace=name_space,
                                           pretty=True, _preload_content=False, follow=follow)
        for content in resp:
            yield content.decode('utf-8')
