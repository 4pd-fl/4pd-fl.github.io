[docs]    def notify(self, engine_id, explore_id, stage, child_stage, content, notice_level=NoticeLevel.INFO.name,
               extra_content=''):
        url = '/notification/{notice_level}'.format(notice_level=notice_level)
        body = {
            'engineId': engine_id,
            'exploreId': explore_id,
            'stage': stage,
            'childStage': child_stage,
            'content': content,
            'message': extra_content,
        }
        return self.request(url, 'post', body=body)
