[docs]    def run(self):
        arg1 = self.job_template
        arg2 = self.job_args
        job_pakage = '.'.join(arg1.split(".")[:-1])
        job_name = arg1.split(".")[-1]
        m1 = importlib.import_module(job_pakage)
        Job= getattr(m1, job_name)
        job_args = json.loads(arg2)
        job = Job(job_id = job_args['job_id'])
        for key in job_args.keys():
            setattr(job,key,job_args[key])
        signal.signal(signal.SIGTERM, job.on_kill)
        signal.signal(signal.SIGINT, job.on_kill)
        context={}
        try:
            job.pre_execute(context)
            job.execute(context)
            job.post_execute(context)
        except Exception  as e:
            log.error("job[{}]运行失败,错误信息:{}",arg1,e)
