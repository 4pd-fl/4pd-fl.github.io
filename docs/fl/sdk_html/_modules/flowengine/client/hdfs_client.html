# -*-coding:utf-8-*-
from hdfs import InsecureClient

from flowengine.config import settings

"""
create a directory and read a file
"""



[docs]class HdfsClient(object):
    """
    能够直接读写 hdfs 文件，原理是封装了 python lib: hdfs 的 InsecureClient 实例。
    """

    def __init__(self):
        url = settings['hdfs_url']
        user_name = settings['hdfs_user']
        self.hdfs_client = InsecureClient(url, user=user_name)
        self.read = lambda file: self.hdfs_client.read(file, encoding='utf-8')
        self.write = lambda file: self.hdfs_client.write(file, encoding='utf-8')
        self.mkdir = self.hdfs_client.makedirs
        self.list = self.hdfs_client.list


[docs]    def get_hdfs_client(self):
        """
        获取 python lib: hdfs 的 InsecureClient 实例
        @return: InsecureClient
        """
        return self.hdfs_client



[docs]    def get_contents(self, file):
        """
        一次性的读取文件的所有内容
        @param file: 文件的 hdfs 路径。
        @return: str
        """
        with self.read(file) as f:
            return f.read()
