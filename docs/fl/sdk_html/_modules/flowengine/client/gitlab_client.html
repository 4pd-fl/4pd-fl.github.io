[docs]    def create_gitlab_project(self, project_name):
        url = "https://gitlab.4pd.io/api/v4/projects"
        import_url = "https://gitlab.4pd.io/flowengine-developer/flowengine-sc/sc-origin"
        data = "name={project_name}&namespace_id=271&import_url={import_url}&" \
               "auto_devops_enabled=true&repository_access_level=enabled&" \
               "visibility=public".format(project_name=project_name, import_url=import_url)
        return requests.post(url, headers=self.headers, data=data)
