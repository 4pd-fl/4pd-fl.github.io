[docs]    def submit_dag(self, dag_json):
        workflow_url = '{}/api/workflow'.format(self.url)
        retries = 3
        ret = None
        for __ in range(retries):
            try:
                if type(dag_json) == str:
                    dag_json = json.loads(dag_json)
                ret = requests.post(workflow_url, None, dag_json, headers=self.headers)
                break
            except Exception as e:
                log.error(e)
        if ret is None:
            raise Exception('failed to submit dag to:{}'.format(self.url))
        return ret

        kvs = {

            'node_id': node_id,

            'log_key': log_key

        workflow_url = '{pre_url}/api/workflow/{dag_id}/run/{run_id}/operator/{node_id}/log/{log_key}'.format(**kvs)
        return requests.get(workflow_url, headers=self.headers)
