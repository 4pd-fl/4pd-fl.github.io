        dag_id, run_id = res_dag['id'], res_dag['runId']
        self.workflow.terminate(dag_id, run_id)

        self.assertEqual(res_info['dagStatus'], 'TERMINATED')



[docs]    def test_wait_for(self):
        res_dag = self.workflow.submit_dag(self.dag_json)
        dag_id = res_dag['id']
        run_id = res_dag['runId']
        res = self.workflow.wait_for(dag_id, run_id)
        self.assertEqual(res['dagId'], dag_id)
        self.assertEqual(res['dagRunId'], run_id)
        self.assertEqual(res['dagStatus'], 'SUCCEEDED')
        self.assertEqual(res['allNodeInfo'][0]['nodeStatus']['scheduleStatus'], 'SUCCEEDED')
        self.assertEqual(res['allNodeInfo'][1]['nodeStatus']['scheduleStatus'], 'SUCCEEDED')
