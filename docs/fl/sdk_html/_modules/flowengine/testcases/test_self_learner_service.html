[docs]    def test_resume_self_learner(self):
        url = self.self_learner.client.url
        headers = self.self_learner.client.headers
        query_url = '{url}/executionHistory/execution/{execution_id}/query'.format(url=url, execution_id=3)
        data = requests.get(query_url, headers=headers)
        result = data.json()
        dag_info = result['data']
        dag_dto = dag_info['dag']
        all_node_info = dag_info['allNodeInfo']
        all_node_info_map = {node['nodeId']: node for node in all_node_info}
        node_list = dag_dto['nodes']
        for node in node_list:
            if node['id'] in all_node_info_map:
                status = all_node_info_map[node['id']]['nodeStatus']['scheduleStatus']
                if status == 'SUCCEEDED':
                    node['status'] = 'SUCCEEDED'
        dag_dto['nodes'] = node_list
        dag_dto['scheduleMessage']=None
        dag_dto['status']='INITIALIZED'

        submit_url = '{url}/workflow/dagPlugin/{execution_id}/submit'.format(url=url, execution_id=3)
        resume_data = requests.post(submit_url, headers=headers, data=json.dumps(dag_dto))
        print(resume_data.status_code)
