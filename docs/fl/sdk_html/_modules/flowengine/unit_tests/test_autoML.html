[docs]    def test_get_file(self):
        client = self.automl.client
        url = '/engines/workflow/automl_explore_lrtrain_python-script_16_3/run/327/operator/Node_FeatureExtract_LR_1/log/application_1553504115883_74704-log'
        with client.request(url, 'get') as r:
            r.raise_for_status()
            with tempfile.NamedTemporaryFile(mode='wb') as f:
                for chunk in r.iter_content(chunk_size=8192):
                    if chunk:  # filter out keep-alive new chunks
                        print('-----')
                        f.write(chunk)
                f.flush()
                with open(f.name, 'r') as fi:
                    for line in fi:
                        print(line)
