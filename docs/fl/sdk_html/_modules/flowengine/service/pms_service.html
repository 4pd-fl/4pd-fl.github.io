[docs]def pms_response_decorator(func):
    @functools.wraps(func)
    def new_f(*args, **kwgs):
        try:
            resp = func(*args, **kwgs)
        except Exception as e:
            msg_head = '内部错误，请联系管理员。详情如下：\n'
            msg = traceback.format_exc()
            raise Exception(msg_head + msg)
        if resp.status_code not in [200, 201, 202]:
            msg_head = '访问异常，reponse info:\n'
            msg = print_response(resp, only_return=True)
            raise Exception(msg_head + msg)
        body = resp.json()
        print(body)
        # print('body:', print_response(resp, only_return=True))
        if 'status' in body:
            if str(body['status']) != 'success':
                msg_head = '访问失败：' + (body['msg'] or '请联系管理员。')
                msg_head += '\nreponse info:\n'
                msg = print_response(resp, only_return=True)
                raise Exception(msg_head + msg)

            if 'count' in body:
                return "{count} alert created".format(count=body['count'])
            elif 'data' in body:
                return body['data']

        elif 'kind' in body:
            if 'items' in body:
                return body['items']
            else:
                return body['metadata']

    return new_f

    PMS 的 service 类，能够进行告警管理。详情参见 PMS 产品介绍相关文档。

    def __init__(self, url=None):

[docs]    @pms_response_decorator
    def get_alert(self, alert_name=None):
        """获取告警信息

        按照 告警名字 获取对应的 告警

        :param str alert_name: 告警名字,

        :return: 告警 的实例

        """
        ret = self.pms_client.get_alert(alert_name)
        vals = ret.json().get('data', [])

        if len(vals) == 0:
            return None
        if len(vals) > 1:
            print('There are more than one alert.')
        return ret
