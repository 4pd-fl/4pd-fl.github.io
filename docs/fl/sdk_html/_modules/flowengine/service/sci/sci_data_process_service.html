        """导入数据预览

        :param str table_file_path: 导入文件的路径

        :param str group_prn: 参见 :py:class:`AutoMLService.import_hdfs`

            .. code-block:: python

                {

                    # 验证类型
                    "validateType": [],

                    # 值类型

        """

                                        table_file_path, group_prn, group_name)

[docs]    @automl_response_decorator
    def import_hdfs(self, engine_key, group_name, upload_config):
        """导入 hdfs 数据

        将 hdfs 地址的数据导入 automl 中，注意该接口只支持导入一个数据文件

        :param str engine_key: 引擎 key
        :param dict upload_config: 导入文件的相关配置，下面配置可直接使用，其他无注释说明则无需修改

            .. code-block:: python

                {   # 列异常行的处理方式，SKIP(抛弃异常行），TERMINATE(引入失败）
                    "columnError": "SKIP",
                    # 类型异常行的处理方式，SKIP(抛弃异常行），NULL(设置为Null)， TERMINATE(引入失败）
                    "typeError": "SKIP",
                    # 高级参数，可自定义，advanced_config = json.dumps({'-m':'4', 'mapreduce.job.reduces':'3'})
                    "config": {
                        # 高级参数默认值
                        "advanced_config": "{\"-m\": \"4\",\"mapreduce.job.reduces\":\"3\", \"mapreduce.map.memory.mb\": \"4096\", \"mapreduce.map.java.opts\": \"-Xmx3072M\", \"mapreduce.reduce.memory.mb\": \"4096\", \"mapreduce.reduce.java.opts\": \"-Xmx3072M\"}"
                    },
                    # 文件类型：parquet（压缩文件），text（文本文件）
                    "fileType": "text",
                    # 数据类型，behaviour（行为表），feedback（反馈表）
                    "dataType": "behaviour",
                    # 编码格式：utf-8，gbk，gb2312
                    "encoding": "utf-8",
                    "timing": False,
                    "partitionField": "",
                    "granularity": "DAY",
                    "dateFormat": None,
                    "dateTimeFormat": None,
                    # label 字段，行为表可置 None
                    "label": None,
                    # 主键字段
                    "primaryKey": "devDid",
                    "colDelimiter": "0x2C",
                    "conf": {},
                    # 行为表/反馈表的数据组 prn，可从 self.get_sys_params 返回值获取
                    # 从返回值获取行为表数据组 prn:list(ret['engineDataInfo']['behaviourDataGroupMap'].values())[0]['groupPrn']
                    # 从返回值获取反馈表数据组 prn:list(ret['engineDataInfo']['feedbackDataGroupMap'].values())[0]['groupPrn']
                    "tableGroupPrn": "automl/qsczmt_behaviourTable_20194921060620.table-group",
                    # 文件首行是否为字段名（schema）
                    "firstLineSchema": True,
                    # 存储名称
                    "name": "hdfs_table_1561433501265",
                    "rowDelimiter": "0x0A",
                    # 数据 schema ，通过 self.preview_data 获取，如不设置则使用 self.preview_data 的 schema
                    "schemas":[
                                {
                                  "name": "col_1",
                                  "type": "Int",
                                  "validateType": [],
                                  "use": true,
                                  "valueType": "DiscreteNum"
                                },
                    ]
                    "storageFormat": "",
                    # 可设置为存储名称（name）
                    "ttable": "hdfs_table_1561433501265",
                    # 待导入的文件的 hdfs 地址
                    "urls": [
                        "hdfs:///automl/sample-imp-pro-13-19-20190506-20190519.csv"
                    ],
                    "importType": "hdfs"
                }

        :return: ``ret['tableId']`` 对应着 :py:class:`AutoMLService.list_job` 返回值的 ``ret['jobList']`` 中的 ``jobPrn``
        :rtype: dict

        """
        # only one url
        logic_engine_id = self.get_logic_engine_id(engine_key)
        # check hdfs
        hdfs_path = upload_config['urls'][0]
        self.check_hdfs(logic_engine_id, hdfs_path)
        # check schema
        group_prn = upload_config['tableGroupPrn']
        data_type = upload_config['dataType']
        if 'schemas' not in upload_config:
            file_type = upload_config['fileType']
            encoding = upload_config['encoding']
            is_first_line_schema = upload_config['firstLineSchema']
            preview_result = self.preview_data(engine_key, data_type=data_type, file_type=file_type, encoding=encoding,
                                               is_first_line_schema=is_first_line_schema, table_file_path=hdfs_path,
                                               group_prn=group_prn, group_name=group_name)
            upload_config['schemas'] = preview_result['schema']
        self.check_schema(logic_engine_id, data_type=data_type, group_prn=group_prn, group_name=group_name, schema=upload_config['schemas'])
        return self.client.import_hdfs(logic_engine_id, group_prn, group_name, upload_config)
