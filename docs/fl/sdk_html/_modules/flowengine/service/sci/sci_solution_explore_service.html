    def create_exploration(self, engine_key, explore_config):
        """创建探索

        配置模型探索，配置完探索并未启动

        :param str engine_key: 引擎 key
        :param dict explore_config: 探索的配置,

            .. code-block:: python

                {
                  # 行为数据反馈数据表的切片配置，key 为 table prn
                  "sliceRangeMap": {
                    # 行为表/反馈表的数据表 prn，可从 self.get_sys_params 返回值获取
                    # 从返回值获取行为表数据组 prn:list(ret['engineDataInfo']['dataClusterMap']['behaviour']['dataGroupMap'].keys()
                    # 表 1 的配置
                    "engine-key_behaviourTable_20193409030712": {
                      # 范围规则，0：全部切片，1：部分切片
                      "rangeRule": 1,
                      # 切片范围，如下表示从 1 到 100
                      "sliceRange": [
                        "1",
                        "100"
                      ]
                    },
                    # 表 2 的配置
                    "engine-key_feedbackTable_20193409030712": {
                      "rangeRule": 0,
                      "sliceRange": [
                        "1",
                        "1000"
                      ]
                    }
                  },
                  # 生产样本参数
                  "instanceGenParams": {},
                  # fe 脚本生成所需参数
                  "feGenParams": {},
                  # 样本自动拆分算子配置
                  "modelingComponentParamMap": {
                    "Node_DataSplit_1": {
                      "method": 0,
                      "splitRatio": 0.8,
                      "isRandom": true,
                      "randomSeed": 1,
                      "isLevel": false,
                      "levelBy": "",
                      "splitRule": "",
                      "orderBy": "",
                      "sortOrie": 1
                    }
                  },
                  # 是否自动部署自学习
                  "autoSelfLearning": false,
                  # 探索退出条件，manual（手动）、timeLimit（时长，单位小时）、aucThreshold（验证auc）、trainTimes（训练轮数）
                  "exitType": "manual",
                  # 探索退出条件的值，如手动退出（manual）可不设。
                  "exitValue": null,
                  # 多表自动拼接
                  "multiTableAutoJoin": true,
                  # 是否使用 DSN（试验功能）
                  "usingDSN": false,
                  # 如样本不平衡，是否自动拆分样本。按某字段拆分也可在 modelingComponentParamMap 设置。
                  "autoInsSplit": false,
                  # 是否使用 LR 增强（试验功能）
                  "usingMultiLR": false,
                  "multiTableGeParams": {
                    "maxEpochs": 1,
                    "maxFeatures": 100
                  }
                }

        :return: ``ret['exploreId']`` 表示探索 ID
        :rtype: dict

        """
        logic_engine_id = self.get_logic_engine_id(engine_key)
        resp = self.client.save_explore_config(logic_engine_id, explore_config)
        return resp

    def start_exploration(self, engine_key, explore_id):
        """开启探索

        开启探索

        :param str engine_key: 引擎 key
        :param int explore_id: 探索 ID，参见 :py:class:`AutoMLService.start_exploration` 返回值

        :return: ``experiment = ret['experiment']，experiment['createTime']`` 表示探索试验开启时间，
            ``experiment['modifyTime']`` 表示最新修改时间。
        :rtype: dict

        """
        logic_engine_id = self.get_logic_engine_id(engine_key)
        resp = self.client.start_explore(logic_engine_id, explore_id)
        return resp



[docs]    def create_and_start_exploration(self, engine_key, explore_config):
        """创建并开启一次探索

        创建并开启一次探索

        :param str engine_key: 引擎 key
        :param dict explore_config: 探索的配置，详情参见 :py:class:`AutoMLService.start_exploration` 参数 ``explore_config``

        :return: 参见 :py:class:`flowengine.service.automl_service.start_exploration` 返回值

        """
        data = self.create_exploration(engine_key, explore_config)
        explore_id = data['exploreId']
        return self.start_exploration(engine_key, explore_id)



[docs]    @automl_response_decorator
    def get_explore_status(self, engine_key, explore_id):
        """获取探索运行信息

        获取探索运行状态和相关信息

        :param str engine_key: 引擎 key
        :param int explore_id: 探索 ID，参见 :py:class:`AutoMLService.start_exploration` 返回值

        :return: ``ret['currentStage']`` 表示当前 ``stage`` ，``ret['currentProgress']`` 当前 ``stage`` 进度，``ret['status']`` 状态信息。

            其中 ``currentStage`` 取值：

                .. code-block:: python

                    ["uncreated", "create", "dataImport", "instanceExplore",
                    "instanceProcess", "modelBuild", "publish"]

            其中 ``status`` 取值：

                .. code-block:: python

                    ['instanceProcess_succ', 'instanceProcess_terminate', 'instanceProcess_running',
                    'instanceProcess_fail', 'modelBuild_ready', 'modelBuild_running', 'modelBuild_terminate',
                    'modelBuild_succ', 'modelBuild_fail', 'publish_succ', 'publish_fail']

        :rtype: dict

        """
        logic_engine_id = self.get_logic_engine_id(engine_key)
        resp = self.client.get_explore_status(logic_engine_id, explore_id)
        return resp
