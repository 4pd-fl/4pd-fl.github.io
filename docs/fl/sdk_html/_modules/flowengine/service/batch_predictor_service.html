[docs]    def get_history(self, solution_key=None, count=1000, offset=0, order_by='start_time', asc=False):
        """获取 solution config

        获取批量预估历史列表

        :param str solution_key: solution key
        :param int count: 数量
        :param int offset: 偏移量
        :param str order_by: 排序字段
        :param bool asc: 是否升序


        :return: 批量预估历史，currentStatus 和 currentStage 请参考对应枚举类
        :rtype: dict

        """
        data = self._get_history(solution_key=solution_key, count=count, offset=offset, order_by=order_by, asc=asc)
        count = data.get('count', 0)
        if count == 0:
            return []
        records = data.get('executionHistories', [])
        for record in records:
            record['currentStatus'] = ExecutionStatus(record['currentStatus']).name
            record['currentStage'] = ExecutionStage(record['currentStage']).name
        return records
