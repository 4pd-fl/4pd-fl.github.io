[docs]    def push_sc_image(self, image_url):
        stream = False
        decode = True
        docker_url = settings['docker_url']
        if image_url.startswith("config-center"):
            ms = MarketService()
            cc_image_url = ms.get_image_from_config_center(image_url.split(":")[1])
            image_url = docker_url + '/' + cc_image_url
        else:
            image_url = docker_url + '/' + image_url
        self.docker_client.push_image(image_url, stream, decode)



[docs]    def save_image(self, sc_key, image_url, root):
        docker_url = settings['docker_url']
        if image_url is None:
            print(colored('this scImage not found', 'red'))
        else:
            try:
                target_file_name = root + '/' + sc_key + '.tar'
                if image_url.startswith("config-center"):
                    ms = MarketService()
                    cc_image_url = ms.get_image_from_config_center(image_url.split(":")[1])
                    image_url = docker_url + '/' + cc_image_url
                else:
                    image_url = docker_url + '/' + image_url
                self.docker_client.save_image(image_url, target_file_name)
            except Exception as e:
                print(colored('SAVING IMAGE FAILED ' + str(e), 'red'))
                pass
