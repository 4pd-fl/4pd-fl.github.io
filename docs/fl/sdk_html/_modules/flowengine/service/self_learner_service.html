[docs]def self_learner_response_decorator(func):
    @functools.wraps(func)
    def new_f(*args, **kwgs):
        try:
            resp = func(*args, **kwgs)
        except Exception as e:
            msg_head = '内部错误，请联系管理员。详情如下：\n'
            msg = traceback.format_exc()
            raise Exception(msg_head + msg)
        if resp.status_code not in [200, 201, 202]:
            msg_head = '访问异常，reponse info:\n'
            msg = print_response(resp, only_return=True)
            raise Exception(msg_head + msg)
        body = resp.json()
        if str(body['baseStatus']) != 'OK':
            msg_head = '访问失败：' + (body['msg'] or '请联系管理员。')
            msg_head += '\nreponse info:\n'
            msg = print_response(resp, only_return=True)
            raise Exception(msg_head + msg)
        return body['data']

    return new_f

[docs]    def get_history(self, solution_key=None, count=1000, offset=0, order_by='start_time', asc=False):
        """获取 solution config

        获取自学习历史列表

        :param str solution_key: solution key
        :param int count: 数量
        :param int offset: 偏移量
        :param str order_by: 排序字段
        :param bool asc: 是否升序


        :return: 自学习历史，currentStatus 和 currentStage 请参考对应枚举类
        :rtype: dict

        """
        data = self._get_history(solution_key=solution_key, count=count, offset=offset, order_by=order_by, asc=asc)
        count = data.get('count', 0)
        if count == 0:
            return []
        records = data.get('executionHistories', [])
        for record in records:
            record['currentStatus'] = ExecutionStatus(record['currentStatus']).name
            record['currentStage'] = ExecutionStage(record['currentStage']).name
        return records



[docs]    @self_learner_response_decorator
    def start_once_self_learner(self, data_sources):
        """开启自学习

        开启一次自学习

        :param list data_sources: 自学习启动配置，格式如下：

            .. code-block:: python

                [ # 行为表配置
                  {
                    "prn": "behavior.table-group",
                    # 全部切片
                    "sliceType": "ALL"
                  },
                  # 反馈表配置
                  {
                    "prn": "feedback.table-group",
                    #按数量范围调取切片
                    "sliceType": "PART",
                    "sliceValue": {
                      "from": 1,
                      "to": 1000
                    }
                  }
                ]

        :return: 开启信息
        :rtype: dict

        """

        data_sources_config = {
            "dataSources": [
                {
                    "id": None,
                    "name": None,
                    "prn": None,
                    "isTableGroup": True,
                    "sliceType": "ALL",
                    "sliceValue": {
                        "from": 1,
                        "to": 1000
                    },
                    "supportTime": False
                },
                {
                    "id": None,
                    "name": None,
                    "prn": None,
                    "isTableGroup": True,
                    "sliceType": "PART",
                    "sliceValue": {
                        "from": 1,
                        "to": 1000
                    },
                    "supportTime": False
                }
            ],
            "params": None
        }

        solution = self.get_solution()
        solution_key = solution.get('solutionKey', None)
        assert solution_key is not None and len(solution_key) > 0, 'solutionKey is required'
        for v in data_sources:
            v.update(id=v['prn'])
        data_sources_config['dataSources'][0].update(data_sources[0])
        data_sources_config['dataSources'][1].update(data_sources[1])
        return self.client.start_once(solution_key, data_sources_config)
