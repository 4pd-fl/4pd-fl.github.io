[docs]    def add(self, trigger_time, cmd, comment='flowengine sdk crontab'):
        """添加定时任务

        添加定时任务到系统的 crontab 中，相当于操作 'crontab -e'

        :param str trigger_time: 任务触发时机，语法和 crontab -e 时间部分一致
        :param str cmd: 任务命令，和 crontab -e 的命令部分一致
        :param str comment: 注释

        :return: None

        """
        trigger_time = ' '.join(trigger_time.split())
        job = self.new(cmd, comment=comment)
        job.setall(trigger_time)
        if not job.is_valid():
            raise Exception('crontab is not valid:%s' % str(job))
        self.write_to_user()



[docs]    def add_function(self, trigger_time, func, comment=None):
        """添加定时任务

        类似于 add 方法，但能根据传入的函数自动生成相关命令。

        :param str trigger_time: 任务触发时机，语法和 crontab -e 时间部分一致
        :param lambda func: 能通过 module 直接 import 的函数（除 lambda 和类中定义的函数）
        :param str comment: 注释

        :return: None

        """
        if not callable(func):
            raise Exception('%s is not callable.' % func)
        if Utils.isalambda(func):
            raise Exception('lambda is not supported now.')
        file_path = inspect.getsourcefile(func)
        file_name = os.path.basename(file_path)
        module_name = file_name.rsplit('.', 1)[0]
        dir_path = os.path.dirname(file_path)
        func_name = func.__name__
        python_bin_path = Utils.which('python3')
        cmd = 'cd {dir_path};{python_bin_path} -c "from {module_name} import {func_name};{func_name}()"'. \
            format(dir_path=dir_path, python_bin_path=python_bin_path, module_name=module_name, func_name=func_name)
        log.info('crontab cmd is:{}'.format(cmd))
        self.add(trigger_time, cmd, comment=comment)
