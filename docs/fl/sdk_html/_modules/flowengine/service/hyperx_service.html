[docs]    def list_self_learning_model(self, engine_key):
        """获取自学习模型列表

        获取所有自学习模型

        :param str engine_key: 引擎 key,

        :return: 内容数据, 如果 ``"currentStatus" == "SUCCESS"`` 并且 ``"currentStage" == "REPORTING"`` 表示成功

            枚举值参见 :py:class:`flowengine.service.self_learner_service.ExecutionStage` 和  :py:class:`flowengine.service.self_learner_service.ExecutionStatus`

            modelApplication 表示用于上线（在线预估、批量预估）的 model prn
        :rtype: list(dict)
        """
        data = self.selfLearner.get_solution()
        solution_key = data['solutionKey']
        models = self.selfLearner.get_history(solution_key)
        report_model_map = self.automl.get_report_model_map(engine_key)
        for model in models:
            model['modelApplication'] = report_model_map.get(model['modelReport'], '')
            model.pop('modelReport', None)
        return models



[docs]    def get_self_learner_model_reports(self, model_prn, engine_key):
        """获取自学习模型的模型报告

        获取 model_prn 对应模型的模型分析报告

        :param str model_prn: 模型 prn ，来自 :py:class:`HyperXService.list_self_learning_model` 的 modelApplication
        :param str engine_key: 引擎 key

        :return: 内容数据, ret['list'][x]['content']['aucMetrics']['globalAuc'] 为测试集 auc
        :rtype: dict

        """
        report_model_map = self.automl.get_report_model_map(engine_key)
        report_prn = None
        for k, v in report_model_map.items():
            if v == model_prn:
                report_prn = k
                break
        if report_prn is None:
            msg = 'catn find the report for model_prn:%s with engine_key:%s' % (model_prn, engine_key)
            log.info(msg)
            return None
        pdms = PdmsService()
        reports = pdms.get_related_model_reports(report_prn)
        return reports
