[docs]    def list_active_apps(self, workspace_id):
        """获取运行中的引擎列表

        根据 workspace_id 获取运行中的引擎

        :param int workspace_id: workspace id
        :return: 引擎列表
        :rtype: list
        """
        try:
            resp = self.engine_client.list_active_apps(workspace_id)
        except IOError:
            raise Exception(resp.text)
        resp_data = resp.json()
        try:
            if 'appList' in resp_data['data']:
                return resp_data['data']['appList']
            else:
                return resp_data['data']
        except IOError:
            raise Exception(resp_data)

[docs]    def list_apps(self, workspace_id):
        """获取引擎列表

        根据 workspace_id 获取所有的引擎

        :param int workspace_id: workspace id
        :return: 引擎列表
        :rtype: list

[docs]    def offline(self, instance_id):
        """停止引擎

        下线指定引擎

        :param int instance_id: 引擎实例 id
        :return: 是否成功
        :rtype: bool
        """
        try:
            resp = self.engine_client.offline(instance_id)
        except IOError:
            raise Exception(resp)
        resp_data = resp.json()
        log.info(resp_data)
        try:
            if resp_data['baseStatus'] == 'OK':
                log.info('Stop Success')
                return resp_data
            else:
                log.info('Stop FAILED')
                return resp_data
        except IOError:
            raise Exception(resp_data)



[docs]    def get_engine_id_by_engine_key(self, workspace_name, engine_key):
        """获取引擎 id

        根据 engine key 获取对应的 engine id

        :param str workspace_name: workspace 名称
        :param str engine_key: 引擎 key
        :return: 引擎 id
        :rtype: int
        """
        keystone = KeystoneService()
        target_workspace = keystone.get_workspace_by_workspace_name(workspace_name)
        workspace_id = target_workspace['id']
        engines = self.list_apps(workspace_id)
        target_engine = next((e for e in engines if e['engineKey'] == engine_key), None)
        if target_engine is None:
            raise Exception('cant find the workspace by the engine key:%s' % engine_key)
        engine_id, url = target_engine['instanceId'], target_engine['url']
        physical_engine_id = int(url.split('/')[2])
        return engine_id, physical_engine_id, engine_key, workspace_id, url
