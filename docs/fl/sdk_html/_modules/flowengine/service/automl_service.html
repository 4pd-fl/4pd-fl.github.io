[docs]def automl_response_decorator(func):
    @functools.wraps(func)
    def new_f(*args, **kwgs):
        try:
            resp = func(*args, **kwgs)
        except Exception as e:
            msg_head = '内部错误，请联系管理员。详情如下：\n'
            msg = traceback.format_exc()
            raise Exception(msg_head + msg)
        if resp.status_code not in [200, 201, 202]:
            msg_head = '访问异常，reponse info:\n'
            msg = print_response(resp, only_return=True)
            raise Exception(msg_head + msg)
        body = resp.json()
        # print('body:', print_response(resp, only_return=True))
        if str(body['status']) != '0':
            msg_head = '访问失败：' + (body['msg'] or '请联系管理员。')
            msg_head += '\nreponse info:\n'
            msg = print_response(resp, only_return=True)
            raise Exception(msg_head + msg)
        return body['data']

    return new_f
