        """提交 dag

        提交 dag 到 pws 上运行

        :param str dag_json: dag 的描述信息

        :return: dag 的详细信息，可参考 :py:class:`WorkFlowService.get_workflow_info` 返回值

[docs]    def wait_for(self, dag_id, run_id):
        """等待 dag 完成

        循环查看 dag 是否完成，并一直等待其完成

        :param str dag_id: dag 的 dag_id
        :param int run_id: dag 的 run_id

        :return: dag 的详细信息，可参考 :py:class:`WorkFlowService.get_workflow_info` 返回值
        :rtype: dict

        """
        while True:
            dag_info = self.get_workflow_info(dag_id, run_id)
            dag = DAG(dag_info)
            status_stat = dag.gather_node_schduler_status()
            if status_stat.get(ScheduleStatus.FAILED.name, []):
                for node in status_stat[ScheduleStatus.FAILED.name]:
                    log.info('node_id:{}, status:{}'.format(dag.get_node_id(node), dag.get_node_status(node)))
                    for entry in dag.get_log_entries(node):
                        node_id = dag.get_node_id(node)
                        log_key = dag.get_log_key(entry)
                        txt = self.get_log(dag_id, run_id, node_id, log_key)
                        log.error(txt)
                    else:
                        log.info('cant get the log key for now.')
                break
            # The left are succeeded or terminated.
            elif dag.is_finished():
                log.info('dag is finished:{}'.format(dag.dag['dagStatus']))
                if dag.get_dag_status() == ScheduleStatus.SUCCEEDED.name:
                    log.info('if succeeded, result prn is:%s' % dag.get_result_prn())
                break
            else:
                log.info(list(map(lambda x: (x[0], len(x[1])), status_stat.items())))
                log.info('next monitor....., sleep 10 sec')
                time.sleep(10)
        return dag_info



[docs]    def submit_and_wait_for(self, dag_json):
        """提交 Dag 并等待起结束

        提交 Dag 并等待起结束

        :param str dag_json: 同 :py:class:`WorkFlowService.submit_dag` 方法

        :return: dag 的详细信息，可参考 :py:class:`WorkFlowService.get_workflow_info` 返回值
        :rtype: dict

        """
        ret = self.submit_dag(dag_json)
        dag_id, run_id = ret['id'], ret['runId']
        return self.wait_for(dag_id, run_id)
