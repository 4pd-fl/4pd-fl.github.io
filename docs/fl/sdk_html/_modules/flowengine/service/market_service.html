[docs]    def get_template_markets(self):
        """获取方案列表

        获取所有方案

        :return: 方案列表
        :rtype: list
        """
        try:
            resp = self.engine_client.get_template_markets()
        except IOError:
            raise Exception(resp)
        resp_data = resp.json()
        log.info(resp_data)
        try:
            if 'data' in resp_data:
                if 'rows' in resp_data['data']['rows']:
                    return resp_data['data']['rows']
        except IOError:
            raise Exception(resp_data)

[docs]    def export_asol(self, asol_id):
        """ 导出 asol

        导出指定id的asol。

        :param asol_id: asol id
        :return: asol 压缩包
        """
        resp = self.market_client.export_asol(asol_id)
        resp.raise_for_status()
        tempstr = resp.headers['Content-Disposition']
        fileName = tempstr[tempstr.find('=') + 1:]  # 获取filename的值
        result = re.findall(r'"([^"]*)"', fileName)[0].encode("ISO8859-1").decode("utf-8")
        with open(result, 'wb') as handle:
            for block in resp.iter_content(1024):
                handle.write(block)

[docs]    def save_dir_as_asol(self, asol_dir):
        try:
            tar = tarfile.open(asol_dir + '.asol', 'w:gz')
            for root, dir, files in os.walk(asol_dir):
                for file in files:
                    fullpath = os.path.join(root, file)
                    tar.add(fullpath)
            tar.close()
        except Exception as e:
            print(colored('ERROR PACKAGE AS ASOL, PLEASE RETRY LATER.', 'red'))
            os._exit(1)
        os.system("rm -rf " + asol_dir)
