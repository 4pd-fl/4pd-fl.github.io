[docs]    def get_instance(self, instance_prn):
        """获取样本表信息

        按照 prn 获取对应的 instance

        :param str instance_prn: instance 的 prn,

        :return: Instance 的实例
        :return: pdms_models.Instance

        """
        ret = self.telamon_client.get_instance(instance_prn)
        ret.raise_for_status()
        vals = ret.json().get('data', {}).get('list', [])
        if len(vals) > 1:
            raise Exception('There are more than one instance')
        if len(vals) == 0:
            return None
        else:
            dict_val = vals[0]
            return Instance(**dict_val)

        """数据引入

        引入数据到 pdms 上

        Args:
            :body (dict): 详细请参见【pdms:数据引入部分】,例如："{'dataSource': {'protocol': 'LOCAL', 'encoding': 'UTF-8', 'fileType': 'text', 'lineSpliter': ';', 'columnSpliter': ',', 'isFirstLineSchema': True, 'url': 'hdfs:///feedback.csv'}, 'dataTarget': {'format': 'parquet', 'typeError': 'SKIP', 'columnError': 'SKIP', 'jsonError': 'SKIP', 'prn': 'flowengine/hdfs_table_154528706848320182420021277.table', 'schema': [{'name': 'col_1', 'type': 'Int'}, {'name': 'automl_id', 'type': 'Int'}]}}".

        Returns:
            dict:包含 Table 的 json 和 其他信息。

        """
        self.check_schemas(body['dataTarget']['schema'])
        ret = self.telamon_client.import_data(body)
        # ret.raise_for_status()
        return ret.json()



[docs]    def check_schemas(self, schemas):
        errors = list()
        for col in schemas:
            if col['type'] in TYPES:
                continue
            else:
                error = "type:%s of col:%s is not invalid, choices:%s" % (col['name'], col['type'], TYPES)
                errors.append(error)
        if len(errors) > 0:
            raise Exception('\n'.join(errors))



[docs]    def import_and_wait(self, body):
        """导入数据并等待

        导入数据并一直阻塞等待，直到完成或者出错。

        :param: 参见 :py:class:`PdmsService.import_data`

        :return: 参见 :py:class:`PdmsService.import_data`

        """
        ret = self.import_data(body)
        prn = ret['data']['table']['prn']
        while True:
            table = self.get_table(prn)
            if table is None:
                pass
            elif table.resourceStatus == ResourceStatus.IMPORTING.name:
                pass
            else:
                break
            log.info('waiting importing, sleep 1 sec.')
            time.sleep(1)
        return table

        """获取所有数据表

        根据 table_group 的 prn 来获取组中的所有数据表

        :param str table_group_prn: table_group 的 prn,

        :rtype: list

    def import_hive_to_pdms(self, hive):
        """hive数据引入到pdms

        hive数据引入到pdms

        :param dict body : 详细请参见【pdms:HiveRequest】,例如：

            .. code-block:: python

               {
                   "dataSource": {
                        "type": "hive",
                        "dbname": "pdms_test",
                        "host": "m7-common-cdh07",
                        "port": "10000",
                        "password": "hive",
                        "username": "hive",
                        "encoding": "utf-8",
                        "table": "student_orc"
                    },
                    "dataTarget": {
                        "schema": [
                            {
                                "name": "id",
                                "type": "Int"
                            },
                            {
                                "name": "name",
                                "type": "String"
                            },
                            {
                                "name": "score",
                                "type": "Double"
                            },
                            {
                                "name": "age",
                                "type": "Int"
                            },
                            {
                                "name": "grade",
                                "type": "Int"
                            }
                        ],
                        "prn": "cb/test2.table",
                    },
                    "splitPartitionedTable": false
                }

        :return: 包含 Table 的 json 和 其他信息。
        :rtype: dict

        """

        tg_prn = hive['dataTarget']['groupPrn']

        if tg_prn:

            table_group = self.get_table_group(tg_prn)

            if table_group is None or tg_prn is None:

                # hive_info = hive['dataSource']

                # resp = self.telamon_client.preview_hive(hive_info, limit=1, random='false')

                schema = hive['dataTarget']['schema']

                tg = TableGroup(prn=tg_prn, schema=schema)

                self.create_table_group(tg.toDict())

                tag = [{
                    'prn': tg_prn,
                    'tag': 'hive_tablegroup'
                }]

                resp_data = self.telamon_client.add_tag_to_tablegroup(tag)
                print(resp_data)

                if 'data' not in resp_data:
                    raise Exception(resp_data)

        resp = self.telamon_client.import_table_from_hive(hive)
        resp_data = resp.json()
        if 'data' in resp_data:
            return resp_data
        else:
            raise Exception(resp_data)
