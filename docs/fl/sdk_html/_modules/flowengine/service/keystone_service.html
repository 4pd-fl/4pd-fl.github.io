from flowengine.client.engine_client import EngineClient

log = logger.getLogger(__name__)

[docs]    def get_workspaces(self):
        """获取 workspace 列表

        获取 keystone 的 workspace 列表

        :return: workspace 列表
        :rtype: list
        """

        resp = self.engine_client.get_workspaces()
        resp_data = resp.json()
        return resp_data['data']['list']



[docs]    def get_users(self, workspace_id):
        """获取 user 列表

        获取 keystone 的某个 workspace 下的 user 列表

        :param int workspace_id: workspace id

        :return: user 列表

        :rtype: list
        """
        try:
            resp = self.engine_client.get_users(workspace_id)
        except IOError:
            raise Exception(resp)
        resp_data = resp.json()
        log.info(resp_data)
        try:
            if 'data' in resp_data:
                if 'list' in resp_data['data']:
                    return resp_data['data']['list']

            else:
                raise Exception(resp_data)
        except IOError:
            raise Exception(resp_data)
            return resp_data



[docs]    def get_workspace_by_workspace_name(self, workspace_name):
        """获取 workpsace 详情

        根据 name 获取 workspace 的详情

        :param str workspace_name: workspace name
        :return: workspace 详情
        :rtype: dict
        """
        workspaces = self.get_workspaces()
        target_workspace = next((w for w in workspaces if w['workspaceName'] == workspace_name), None)
        return target_workspace



[docs]    def get_workspace_id(self, workspace_name):
        """获取 workpsace id

        根据 name 获取 workspace 的 id

        :param str workspace_name: workspace name
        :return: workspace id
        :rtype: int
        """
        workspace = self.get_workspace_by_workspace_name(workspace_name)
        if workspace is None:
            raise Exception('cant find the workspace by the name:%s' % workspace_name)
        return workspace['id']
