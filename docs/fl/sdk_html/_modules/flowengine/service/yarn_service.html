from flowengine.client import YarnClient
from flowengine.config import settings
from flowengine.logger_manager import logger
from flowengine.utils.http_request import response_decorator

log = logger.getLogger(__name__)



[docs]class YarnService(object):
    """
    Yarn 的 service 类，监控集群信息和杀死任务。

    .. todo::
        添加 Yarn 的提交任务功能，目前可用 shell 命令代替。

    """

    def __init__(self):
        url = settings['yarn_url']
        user_name = settings['yarn_user']
        self.client = YarnClient(url, user_name)

    def get_app(self, app_id):
        """获取 app 的详细信息

        获取 app_id 的 app 信息

        :param str app_id: yarn 的 app id,

        :return: app 信息，类似：{'app': {'id':xx}}
        :rtype: dict

        """

    def get_stat(self, app_id):
        """获取 app 的状态

        获取 app 的状态，包含：“NEW”, “NEW_SAVING”, “SUBMITTED”, “ACCEPTED”, “RUNNING”, “FINISHED”, “FAILED”, “KILLED”

        :param str app_id: yarn 的 app id,

        :return: app 的状态，类似：{'state': 'FINISHED'}
        :rtype: dict

        """
        return self.client.get_stat(app_id)
