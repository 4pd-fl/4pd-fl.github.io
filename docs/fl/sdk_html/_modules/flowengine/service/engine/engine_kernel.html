    def __init__(self, headers=None, instance_id=None):
        super().__init__()

            self.url = settings['engine_kernel']
        self.client = EngineKernelClient(self.url, self.headers)

    def start_online_predictor(self, model_prn, engine_key, cpu=1, mem=1024, replica=1, feature_enabled=False,
                               kafka_end_point=None, kafka_topic=None, kafka_interval=0, externel_config={}):
        """部署在线预估

        根据 model_prn 部署批量预估


        :param str model_prn: model prn 地址
        :param str engine_key: 引擎 key
        :param int cpu: 单实例资源配置, cpu 核心数
        :param int mem: 单实例资源配置, 内存（M）
        :param int replica: 实例数量设置
        :param bool feature_enabled: 是否开启自动回流，如开启，kafka 相关参数必须
        :param str kafka_end_point: Kafka 地址
        :param str kafka_topic: Kafka Topic
        :param int kafka_interval: Kafka 回流周期，单位小时
        :param externel_config: map 包含额外参数的配置，配置部分如下：
            :int jvmMemoryLimit: jvm的大小
            :bool isTFEEnabled: 是否开启TFE预估，如开启，zookeeper和主表相关参数必须
            :str feMainTable: TFE预估主表
            :str zkEndPoint: zookeeper addr
            :str zkRootPath: zookeeper的rtidb path
            :dict externalMap: 可额外传递的预估环境变量

        :return: ``true`` 表示部署成功
        :rtype: str

        """
        if externel_config is None:
            externel_config = {}
        config = {
            "params": {
                "cpuLimit": cpu,
                "memLimit": mem,
                "replica": replica,
                "featureLogEnabled": feature_enabled,
                "kafkaEndPoint": kafka_end_point,
                "kafkaTopic": kafka_topic,
                "kafkaInterval": kafka_interval
            },
            "modelDto": {
                "modelPrn": model_prn
            }
        }
        config["params"].update(externel_config)
        loginc_engine_id = self.get_logic_engine_id(engine_key)
        return self.client.post_online_predict_config(loginc_engine_id, config)

[docs]    def create_or_update_pipeline(self, new_pipeline_raw, pipeline_id=None, engine_template_id=1):
        """新增或者更新已有的pipeline

        :param new_pipeline_raw: pipeline定义
        :param pipeline_id: 更新时提供更新的pipeline_id
        :param engine_template_id:引擎模版id，默认为1

        :return:操作状态
        """
        template = self.client.get_template(engine_template_id)
        payload = {}
        payload['engineDashBoardTemplate'] = template['engineDashBoardTemplate']
        payload['indexTemplate'] = template['indexTemplate']
        oldPipelines = template['jobPipelineTemplates']
        new_pipeline = {}
        new_pipeline['engineTemplateId'] = engine_template_id
        new_pipeline['data'] = new_pipeline_raw
        if pipeline_id is not None:
            new_pipeline['id'] = pipeline_id
            for p in oldPipelines:
                if p['id'] == pipeline_id:
                    oldPipelines.remove(p)
                    break
        oldPipelines.append(new_pipeline)
        payload['jobPipelineTemplates'] = oldPipelines
        response = self.client.save_template(
            json.dumps(payload, default=lambda o: o.__dict__, sort_keys=True, indent=6), 'solution-job-runner',
            engine_template_id)
        if response.status_code == 200:
            return json.loads(response.content)
        else:
            return {'status': -1, 'msg': 'post error,try again!'}
